The Reverse Multiple Problem

Find an \(n\)-digit integer \(N\) such that it is equal to an integer multiplier \(x\) times its own reverse.

\(N = x \cdot \text{rev}(N)\)
Example

If we let \(n = 5\) and \(x = 4\), the number is 87,912 because:
87,912 = 4 × 21,978

                <script>
                    document.addEventListener('DOMContentLoaded', () => {
                        document.querySelectorAll('.break').forEach(el => {
                            el.style.height = `${el.dataset.brLength}`
                        });

                        questionOneSubmit()
                    });

                    function questionOneSubmit() {
                        btn = document.getElementById('questionOneBtn');
                        questionOneResult = document.getElementById('questionOneResult')


                        btn.addEventListener("click", () => {
                            x = parseInt(document.getElementById('xOneInput').value);
                            n = parseInt(document.getElementById('nOneInput').value);
                            solutions = []

                            for (num = Math.pow(10, n - 1); num <= Math.pow(10, n) - 1; num++) {
                                const num_reversed = parseInt(num.toString().split('').reverse().join('')) 
                                if (num === x * num_reversed) {
                                    solutions.push(num);
                                    console.log(num);
                                }
                            }

                            console.log(`Result = ${solutions}`)

                            questionOneResult.innerHTML = solutions.length > 0
                                ? solutions.join(', ')
                                : "No solutions";
                        });
                    }
